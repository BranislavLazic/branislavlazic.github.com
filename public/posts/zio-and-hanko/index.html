ZIO and Hanko

March 4, 2025

Enter Hanko#

Hanko is an authorization server available as a cloud solution on hanko.io. It can also be self-hosted for greater control and customization. A key feature of Hanko is authorization via passcodes, along with easy integration with SSO providers such as Google and GitHub.

Essentially, Hanko issues a JWT token, which is stored by the client and sent to a web service for verification. Setting aside the client-side token handling (which can be easily adapted to React), let’s focus on how we can efficiently verify tokens in a Scala web service powered by ZIO.

Naive ZIO implementation#

import cats.implicits.*
import com.auth0.jwk.{Jwk, UrlJwkProvider}
import com.auth0.jwt.JWT
import com.auth0.jwt.algorithms.Algorithm
import com.auth0.jwt.interfaces.DecodedJWT
import io.circe.*
import io.circe.generic.auto.*
import io.circe.parser.decode
import zeitkraft.config.AppConfig
import zeitkraft.server.{ExternalUserId, Principal, ServiceError, UnauthorizedError}
import zio.{Ref, Task, ZIO, ZLayer}

import java.nio.charset.StandardCharsets
import java.security.interfaces.RSAPublicKey
import java.util.Base64

final case class Email(address: String)
final case class HankoPayload(email: Email, sub: String):
  def toPrincipal: Principal = Principal(ExternalUserId(sub), email.address)

final case class HankoAuthServiceLive(appConfig: AppConfig, jwkCache: Ref[Option[Jwk]]) extends AuthService:
  private val provider = new UrlJwkProvider(appConfig.hankoConfig.url)

  /**
   * Introspect will decode the token and return the principal build on the
   * basis of the token claims. The JWK key is cached and refreshed if the token
   * verification fails.
   */
  override def introspect(token: String): Task[Either[ServiceError, Principal]] =
    for {
      decodedJWT <- ZIO.attempt(JWT.decode(token))
      verifyJwtDecoded = verifyJwk(decodedJWT, _)
      jwk <- retrieveJwk(decodedJWT)
      _   <- verifyJwtDecoded(jwk).foldZIO(_ => refreshJwk(decodedJWT).flatMap(verifyJwtDecoded), _ => ZIO.unit)
      decodedPayload = new String(Base64.getDecoder.decode(decodedJWT.getPayload), StandardCharsets.UTF_8)
      result <- ZIO.succeed(
        decode[HankoPayload](decodedPayload).map(_.toPrincipal).leftMap(_ => UnauthorizedError.empty),
      )
    } yield result

  private def verifyJwk(decodedJWT: DecodedJWT, jwk: Jwk) =
    ZIO
      .attempt(
        Algorithm.RSA256(jwk.getPublicKey.asInstanceOf[RSAPublicKey], null).verify(decodedJWT),
      )

  private def retrieveJwk(jwt: DecodedJWT): Task[Jwk] =
    for {
      cachedValue <- jwkCache.get
      jwk         <- cachedValue.map(ZIO.succeed).getOrElse(refreshJwk(jwt) <* ZIO.logInfo("Refreshing JWK"))
    } yield jwk

  private def refreshJwk(jwt: DecodedJWT): Task[Jwk] =
    ZIO.attempt(provider.get(jwt.getKeyId)).tap(jwk => jwkCache.update(_ => Some(jwk)))

object HankoAuthServiceLive:
  val layer: ZLayer[AppConfig & Ref[Option[Jwk]], Nothing, HankoAuthServiceLive] =
    ZLayer.fromFunction(HankoAuthServiceLive.apply)
